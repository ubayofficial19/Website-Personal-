let cart = [];
let currentFilter = 'all';
let selectedApp = null;

// Initialize the app
document.addEventListener('DOMContentLoaded', function() {
renderApps();
setupEventListeners();
});

function setupEventListeners() {
// Mobile menu toggle
document.getElementById('mobileMenuBtn').addEventListener('click', function() {
const menu = document.getElementById('mobileMenu');
menu.classList.toggle('hidden');
});

// Cart button
document.getElementById('cartBtn').addEventListener('click', function() {
showCartModal();
});

// Search functionality
document.getElementById('searchInput').addEventListener('input', function(e) {
searchApps(e.target.value);
});

// Close modals when clicking outside
document.addEventListener('click', function(e) {
if (e.target.classList.contains('modal')) {
closeModal();
closeCartModal();
}
});
}

function renderApps(filteredApps = apps) {
const grid = document.getElementById('appsGrid');
grid.innerHTML = '';

filteredApps.forEach(app => {
const appCard = createAppCard(app);
grid.appendChild(appCard);
});
}

function createAppCard(app) {
const card = document.createElement('div');
card.className = 'bg-white dark:bg-gray-800 rounded-xl shadow-lg card-hover cursor-pointer';
card.onclick = () => showAppModal(app);

card.innerHTML = `
<div class="relative">
<img src="${app.image}" alt="${app.name}" class="w-full h-48 object-cover rounded-t-xl">
<div class="absolute top-4 left-4">
<span class="bg-primary text-white px-3 py-1 rounded-full text-sm font-medium capitalize">
${getCategoryName(app.category)}
</span>
</div>
</div>
<div class="p-6">
<h4 class="text-xl font-semibold text-gray-800 dark:text-white mb-2">${app.name}</h4>
<p class="text-gray-600 dark:text-gray-300 mb-4 line-clamp-2">${app.description}</p>
<div class="flex items-center justify-between">
<span class="text-2xl font-bold text-primary">Rp ${app.price.toLocaleString('id-ID')}</span>
<button onclick="event.stopPropagation(); addToCart(${app.id})"
class="bg-primary text-white px-4 py-2 rounded-lg hover:bg-primary-dark transition-colors">
<i class="fas fa-cart-plus"></i>
</button>
</div>
<div class="flex items-center mt-3">
<div class="flex text-yellow-400 mr-2">
${'<i class="fas fa-star"></i>'.repeat(5)}
</div>
<span class="text-gray-600 dark:text-gray-300 text-sm">(4.8)</span>
</div>
</div>
`;

return card;
}

function getCategoryName(category) {
const names = {
'productivity': 'Produktivitas',
'design': 'Design',
'business': 'Bisnis',
'gaming': 'Gaming'
};
return names[category] || category;
}

function filterApps(category) {
currentFilter = category;

// Update filter buttons
document.querySelectorAll('.filter-btn').forEach(btn => {
btn.classList.remove('active', 'bg-primary', 'text-white');
btn.classList.add('bg-white', 'dark:bg-gray-700', 'text-gray-700', 'dark:text-gray-300');
});

event.target.classList.add('active', 'bg-primary', 'text-white');
event.target.classList.remove('bg-white', 'dark:bg-gray-700', 'text-gray-700', 'dark:text-gray-300');

// Filter and render apps
const filteredApps = category === 'all' ? apps : apps.filter(app => app.category === category);
renderApps(filteredApps);
}

function searchApps(query) {
const filteredApps = apps.filter(app => {
const matchesSearch = app.name.toLowerCase().includes(query.toLowerCase()) ||
app.description.toLowerCase().includes(query.toLowerCase());
const matchesCategory = currentFilter === 'all' || app.category === currentFilter;
return matchesSearch && matchesCategory;
});
renderApps(filteredApps);
}

function showAppModal(app) {
selectedApp = app;
document.getElementById('modalTitle').textContent = app.name;
document.getElementById('modalImage').src = app.image;
document.getElementById('modalCategory').textContent = getCategoryName(app.category);
document.getElementById('modalDescription').textContent = app.description;
document.getElementById('modalPrice').textContent = `Rp ${app.price.toLocaleString('id-ID')}`;

const featuresList = document.getElementById('modalFeatures');
featuresList.innerHTML = '';
app.features.forEach(feature => {
const li = document.createElement('li');
li.className = 'flex items-center text-gray-600 dark:text-gray-300';
li.innerHTML = `<i class="fas fa-check text-green-500 mr-3"></i>${feature}`;
featuresList.appendChild(li);
});

document.getElementById('appModal').classList.add('active');
}

function closeModal() {
document.getElementById('appModal').classList.remove('active');
selectedApp = null;
}

function addToCart(appId) {
const app = apps.find(a => a.id === appId);
if (app) {
const existingItem = cart.find(item => item.id === appId);
if (existingItem) {
existingItem.quantity += 1;
} else {
cart.push({ ...app, quantity: 1 });
}
updateCartBadge();
showNotification('Aplikasi berhasil ditambahkan ke keranjang!');
}
}

function addToCartFromModal() {
if (selectedApp) {
addToCart(selectedApp.id);
closeModal();
}
}

function updateCartBadge() {
const badge = document.getElementById('cartBadge');
const totalItems = cart.reduce((sum, item) => sum + item.quantity, 0);
badge.textContent = totalItems;
badge.classList.add('cart-badge');
setTimeout(() => badge.classList.remove('cart-badge'), 500);
}

function showCartModal() {
renderCartItems();
document.getElementById('cartModal').classList.add('active');
}

function closeCartModal() {
document.getElementById('cartModal').classList.remove('active');
}

function renderCartItems() {
const container = document.getElementById('cartItems');
const totalElement = document.getElementById('cartTotal');

if (cart.length === 0) {
container.innerHTML = '<p class="text-center text-gray-500 dark:text-gray-400 py-8">Keranjang Anda kosong</p>';
totalElement.textContent = 'Rp 0';
return;
}

container.innerHTML = '';
let total = 0;

cart.forEach(item => {
const itemTotal = item.price * item.quantity;
total += itemTotal;

const cartItem = document.createElement('div');
cartItem.className = 'flex items-center space-x-4 p-4 bg-gray-50 dark:bg-gray-700 rounded-lg';
cartItem.innerHTML = `
<img src="${item.image}" alt="${item.name}" class="w-16 h-16 object-cover rounded-lg">
<div class="flex-1">
<h5 class="font-semibold text-gray-800 dark:text-white">${item.name}</h5>
<p class="text-gray-600 dark:text-gray-300">Rp ${item.price.toLocaleString('id-ID')}</p>
</div>
<div class="flex items-center space-x-2">
<button onclick="updateCartItemQuantity(${item.id}, -1)" class="w-8 h-8 bg-gray-200 dark:bg-gray-600 rounded-full flex items-center justify-center hover:bg-gray-300 dark:hover:bg-gray-500">
<i class="fas fa-minus text-sm"></i>
</button>
<span class="text-gray-800 dark:text-white font-medium w-8 text-center">${item.quantity}</span>
<button onclick="updateCartItemQuantity(${item.id}, 1)" class="w-8 h-8 bg-gray-200 dark:bg-gray-600 rounded-full flex items-center justify-center hover:bg-gray-300 dark:hover:bg-gray-500">
<i class="fas fa-plus text-sm"></i>
</button>
</div>
<button onclick="removeFromCart(${item.id})" class="text-red-500 hover:text-red-700 p-2">
<i class="fas fa-trash"></i>
</button>
`;
container.appendChild(cartItem);
});

totalElement.textContent = `Rp ${total.toLocaleString('id-ID')}`;
}

function updateCartItemQuantity(itemId, change) {
const item = cart.find(item => item.id === itemId);
if (item) {
item.quantity += change;
if (item.quantity <= 0) {
removeFromCart(itemId);
return;
}
updateCartBadge();
renderCartItems();
}
}

function removeFromCart(itemId) {
cart = cart.filter(item => item.id !== itemId);
updateCartBadge();
renderCartItems();
}

function checkout() {
if (cart.length === 0) {
showNotification('Keranjang Anda kosong!');
return;
}

const total = cart.reduce((sum, item) => sum + (item.price * item.quantity), 0);
showConfirmDialog(
`Konfirmasi pembelian dengan total Rp ${total.toLocaleString('id-ID')}?`,
() => {
// Simulate checkout process
showNotification('Pembelian berhasil! Terima kasih telah berbelanja di Baystore.');
cart = [];
updateCartBadge();
closeCartModal();
}
);
}

function showNotification(message) {
const notification = document.createElement('div');
notification.className = 'fixed top-20 right-4 bg-green-500 text-white px-6 py-3 rounded-lg shadow-lg z-50 transform translate-x-full transition-transform duration-300';
notification.textContent = message;
document.body.appendChild(notification);

setTimeout(() => {
notification.classList.remove('translate-x-full');
}, 100);

setTimeout(() => {
notification.classList.add('translate-x-full');
setTimeout(() => {
document.body.removeChild(notification);
}, 300);
}, 3000);
}

function showConfirmDialog(message, onConfirm) {
const modal = document.createElement('div');
modal.className = 'fixed inset-0 bg-black bg-opacity-50 flex items-center justify-center z-50';
modal.innerHTML = `
<div class="bg-white dark:bg-gray-800 p-6 rounded-lg shadow-lg max-w-sm w-full mx-4">
<p class="text-gray-700 dark:text-gray-300 mb-4">${message}</p>
<div class="flex justify-end space-x-3">
<button onclick="this.closest('.fixed').remove()" class="px-4 py-2 text-gray-600 dark:text-gray-400 hover:bg-gray-100 dark:hover:bg-gray-700 rounded">Batal</button>
<button onclick="this.closest('.fixed').remove(); (${onConfirm})()" class="px-4 py-2 bg-primary text-white hover:bg-primary-dark rounded">Konfirmasi</button>
</div>
</div>
`;
document.body.appendChild(modal);
}
